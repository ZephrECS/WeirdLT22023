            updateThemeButtons(savedTheme);

        // ============================================
        // FAVORITES SYSTEM - Saves to localStorage
        // ============================================
        
        function getFavorites() {
            const saved = localStorage.getItem('gamerz-favorites');
            return saved ? JSON.parse(saved) : [];
        }

        function saveFavorites(favorites) {
            localStorage.setItem('gamerz-favorites', JSON.stringify(favorites));
        }

        function toggleFavorite(gameTitle) {
            let favorites = getFavorites();
            const index = favorites.indexOf(gameTitle);
            
            if (index > -1) {
                favorites.splice(index, 1);
            } else {
                favorites.push(gameTitle);
            }
            
            saveFavorites(favorites);
            renderGames();
        }

        function isFavorited(gameTitle) {
            return getFavorites().includes(gameTitle);
        }

        // ============================================
        // RENDER GAMES - Automatically sorted by favorites
        // ============================================
        
        function renderGames() {
            const gamesGrid = document.getElementById('gamesGrid');
            gamesGrid.innerHTML = '';
            
            // Sort games: favorites first, then alphabetically
            const sortedGames = [...games].sort((a, b) => {
                const aFav = isFavorited(a.title);
                const bFav = isFavorited(b.title);
                
                if (aFav && !bFav) return -1;
                if (!aFav && bFav) return 1;
                return a.title.localeCompare(b.title);
            });
            
            sortedGames.forEach(game => {
                const card = document.createElement('div');
                card.className = 'game-card';
                card.setAttribute('data-title', game.title.toLowerCase());
                
                const favorited = isFavorited(game.title);
                
                card.innerHTML = `
                    <div class="game-card-image">
                        <img src="${game.image}" alt="${game.title}">
                    </div>
                    <div class="game-card-info">
                        <div class="game-card-title">${game.title}</div>
                        <div class="favorite-btn ${favorited ? 'favorited' : ''}" data-game="${game.title}">
                            ${favorited ? '‚≠ê' : '‚òÜ'}
                        </div>
                    </div>
                `;
                
                // Add click handler for the game card (but not the favorite button)
                const cardImage = card.querySelector('.game-card-image');
                
                cardImage.onclick = () => loadGame(game.gameUrl);
                
                // Add click handler for favorite button
                const favBtn = card.querySelector('.favorite-btn');
                favBtn.onclick = (e) => {
                    e.stopPropagation();
                    toggleFavorite(game.title);
                };
                
                gamesGrid.appendChild(card);
            });
        }

        // Initial render
        renderGames();

        // Function to load game from GitHub
        async function loadGame(gameUrl) {
            if (!gameUrl) {
                alert('Please add a GitHub RAW game URL to this game card in the code!');
                return;
            }
            
            try {
                const response = await fetch(gameUrl);
                const code = await response.text();
                
                const newTab = window.open('', '_blank');
                newTab.document.open();
                newTab.document.write(code);
                newTab.document.close();
            } catch (error) {
                alert('Error loading game: ' + error.message);
            }
        }

        // Search functionality
        const searchInput = document.getElementById('searchInput');
        searchInput.addEventListener('input', (e) => {
            const searchTerm = e.target.value.toLowerCase();
            const gameCards = document.querySelectorAll('.game-card');
            
            gameCards.forEach(card => {
                const title = card.getAttribute('data-title');
                if (title.includes(searchTerm)) {
                    card.classList.remove('hidden');
                } else {
                    card.classList.add('hidden');
                }
            });
